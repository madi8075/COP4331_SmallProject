# Create Users Table, and Contacts Table for Local Testing w/ StarfinderDB

CREATE TABLE `StarfinderDB`.`Users` ( `ID` INT NOT NULL AUTO_INCREMENT , `DateCreated` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , `DateLastLoggedIn` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , `FirstName` VARCHAR(50) NOT NULL DEFAULT '' , `LastName` VARCHAR(50) NOT NULL DEFAULT '' , `Login` VARCHAR(50) NOT NULL DEFAULT '' , `Password` VARCHAR(255) NOT NULL DEFAULT '' , PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `StarfinderDB`.`Contacts` ( `ID` INT NOT NULL AUTO_INCREMENT , `FirstName` VARCHAR(50) NOT NULL DEFAULT '' ,`LastName` VARCHAR(50) NOT NULL DEFAULT '' ,`Email` VARCHAR(126) NOT NULL DEFAULT '' ,`Phone` VARCHAR(60) NOT NULL DEFAULT '', `UserID` INT NOT NULL DEFAULT '0' ,`DateCreated` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`ID`)) ENGINE = InnoDB;
